🧭 2. navigation/AppNavigation.kt
package com.tuapp.videoclub.navigation

import androidx.compose.runtime.Composable
import androidx.navigation.NavHostController
import androidx.navigation.compose.NavHost
import androidx.navigation.compose.rememberNavController
import androidx.navigation.compose.composable
import com.tuapp.videoclub.screens.*

@Composable
fun AppNavigation() {
    val navController: NavHostController = rememberNavController()
    NavHost(navController = navController, startDestination = "login") {
        composable("login") { LoginScreen(navController) }
        composable("register") { RegisterScreen(navController) }
        composable("movies") { MovieListScreen(navController) }
        composable("movieDetail/{movieId}") { backStackEntry ->
            MovieDetailScreen(
                navController,
                movieId = backStackEntry.arguments?.getString("movieId") ?: ""
            )
        }
        composable("rentals") { MyRentalsScreen(navController) }
    }
}

package com.tuapp.videoclub.repository

import com.tuapp.videoclub.model.Pelicula
import java.util.*

object FakeMovieRepository {
    private val peliculas = mutableListOf(
        Pelicula(
            id = UUID.randomUUID().toString(),
            titulo = "Matrix",
            descripcion = "Un clásico de ciencia ficción.",
            imagenUrl = "https://i.imgur.com/YOz8EpO.jpg"
        ),
        Pelicula(
            id = UUID.randomUUID().toString(),
            titulo = "Inception",
            descripcion = "Sueños dentro de sueños.",
            imagenUrl = "https://i.imgur.com/NX3zP9c.jpg"
        ),
        Pelicula(
            id = UUID.randomUUID().toString(),
            titulo = "John Wick",
            descripcion = "Acción sin parar.",
            imagenUrl = "https://i.imgur.com/6Ej2t8Q.jpg"
        )
    )

    fun getPeliculas() = peliculas

    fun getPeliculaPorId(id: String): Pelicula? = peliculas.find { it.id == id }

    fun alquilarPelicula(id: String) {
        getPeliculaPorId(id)?.alquilada = true
    }

    fun devolverPelicula(id: String) {
        getPeliculaPorId(id)?.alquilada = false
    }

    fun getPeliculasAlquiladas(): List<Pelicula> = peliculas.filter { it.alquilada }
}

@Composable
fun LoginScreen(navController: NavHostController) {
    var email by remember { mutableStateOf("") }
    var password by remember { mutableStateOf("") }

    Column(modifier = Modifier.padding(16.dp)) {
        Text("Iniciar Sesión", style = MaterialTheme.typography.h5)
        OutlinedTextField(value = email, onValueChange = { email = it }, label = { Text("Email") })
        OutlinedTextField(value = password, onValueChange = { password = it }, label = { Text("Contraseña") })

        Button(onClick = { navController.navigate("movies") }, modifier = Modifier.padding(top = 16.dp)) {
            Text("Entrar")
        }

        TextButton(onClick = { navController.navigate("register") }) {
            Text("No tienes cuenta? Regístrate")
        }
    }
}

🖼 MovieDetailScreen.kt
@Composable
fun MovieDetailScreen(navController: NavHostController, movieId: String) {
    val pelicula = FakeMovieRepository.getPeliculaPorId(movieId)

    pelicula?.let {
        Column(modifier = Modifier.padding(16.dp)) {
            AsyncImage(
                model = it.imagenUrl,
                contentDescription = it.titulo,
                modifier = Modifier
                    .fillMaxWidth()
                    .height(200.dp),
                contentScale = ContentScale.Crop
            )

            Text(it.titulo, style = MaterialTheme.typography.h5)
            Text(it.descripcion)

            Button(onClick = {
                if (it.alquilada) {
                    FakeMovieRepository.devolverPelicula(it.id)
                } else {
                    FakeMovieRepository.alquilarPelicula(it.id)
                }
                navController.popBackStack()
            }, modifier = Modifier.padding(top = 16.dp)) {
                Text(if (it.alquilada) "Devolver" else "Alquilar")
            }
        }
    }
}

✅ 3. Agrupar por categoría en MovieListScreen.kt
@Composable
fun MovieListScreen(navController: NavHostController) {
    val peliculas = FakeMovieRepository.getPeliculas()

    // Agrupamos por categoría
    val peliculasPorCategoria = peliculas.groupBy { it.categoria }

    Column(modifier = Modifier.padding(16.dp)) {
        Text("Películas por Categoría", style = MaterialTheme.typography.h5)

        LazyColumn {
            peliculasPorCategoria.forEach { (categoria, lista) ->
                item {
                    Text(
                        text = categoria,
                        style = MaterialTheme.typography.h6,
                        modifier = Modifier.padding(vertical = 8.dp)
                    )
                }

                items(lista) { pelicula ->
                    MovieCard(pelicula = pelicula) {
                        navController.navigate("movieDetail/${pelicula.id}")
                    }
                }
            }
        }

        Button(
            onClick = { navController.navigate("rentals") },
            modifier = Modifier
                .padding(top = 16.dp)
                .fillMaxWidth()
        ) {
            Text("Mis Alquileres")
        }
    }
}

✅ 4. Asegúrate de tener MovieCard.kt así:
package com.tuapp.videoclub.ui.components

import androidx.compose.foundation.Image
import androidx.compose.foundation.clickable
import androidx.compose.foundation.layout.*
import androidx.compose.material3.*
import androidx.compose.runtime.Composable
import androidx.compose.ui.Modifier
import androidx.compose.ui.layout.ContentScale
import androidx.compose.ui.unit.dp
import coil.compose.AsyncImage
import com.tuapp.videoclub.model.Pelicula

@Composable
fun MovieCard(pelicula: Pelicula, onClick: () -> Unit) {
    Card(
        modifier = Modifier
            .fillMaxWidth()
            .padding(vertical = 8.dp)
            .clickable { onClick() }
    ) {
        Row(modifier = Modifier.padding(8.dp)) {
            AsyncImage(
                model = pelicula.imagenUrl,
                contentDescription = pelicula.titulo,
                modifier = Modifier
                    .size(100.dp)
                    .padding(end = 8.dp),
                contentScale = ContentScale.Crop
            )
            Column {
                Text(pelicula.titulo, style = MaterialTheme.typography.titleMedium)
                Text(pelicula.descripcion, style = MaterialTheme.typography.bodySmall)
                Text("Categoría: ${pelicula.categoria}", style = MaterialTheme.typography.bodySmall)
                if (pelicula.alquilada) {
                    Text("Alquilada ✅", color = MaterialTheme.colorScheme.primary)
                }
            }
        }
    }
}
